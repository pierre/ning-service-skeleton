17  package com.ning.jetty.core.listeners;
18  
19  import com.google.inject.Guice;
20  import com.google.inject.Injector;
21  import com.google.inject.Module;
22  import com.google.inject.Stage;
23  import com.google.inject.servlet.GuiceServletContextListener;
24  import org.apache.commons.lang.StringUtils;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import javax.servlet.ServletContextEvent;
29  
30  /**
31   * Start up the base module when the server comes up. This gets configured in web.xml:
32   * <p/>
33   * <pre>
34   * &lt;context-param&gt;
35   *   &lt;param-name&gt;guiceModuleClassName&lt;/param-name&gt;
36   *   &lt;param-value&gt;com.ning.jetty.core.listeners.SetupServer&lt;/param-value&gt;
37   *  &lt;/context-param&gt;
38   * </pre>
39   */
40  public class SetupServer extends GuiceServletContextListener
41  {
42      private static final Logger log = LoggerFactory.getLogger(SetupServer.class);
43  
44      protected Module guiceModule = null;
45  
46      @Override
47      public void contextInitialized(final ServletContextEvent event)
48      {
49          if (guiceModule != null) {
50              // Overriden in subclasses?
51              super.contextInitialized(event);
52              return;
53          }
54  
55          final String moduleClassName = event.getServletContext().getInitParameter("guiceModuleClassName");
56          if (StringUtils.isEmpty(moduleClassName)) {
57              throw new IllegalStateException("Missing parameter for the base Guice module!");
58          }
59  
60          try {
61              final Class<?> moduleClass = Class.forName(moduleClassName);
62              if (!Module.class.isAssignableFrom(moduleClass)) {
63                  throw new IllegalStateException(String.format("%s exists but is not a Guice Module!", moduleClassName));
64              }
65  
66              guiceModule = (Module) moduleClass.newInstance();
67              log.info("Instantiated " + moduleClassName + " as the main guice module.");
68          }
69          catch (ClassNotFoundException cnfe) {
70              throw new IllegalStateException(cnfe);
71          }
72          catch (InstantiationException ie) {
73              throw new IllegalStateException(ie);
74          }
75          catch (IllegalAccessException iae) {
76              throw new IllegalStateException(iae);
77          }
78  
79          super.contextInitialized(event);
80      }
81  
82      /**
83       * Do *not* use this method to retrieve the injector. It actually creates a new instance
84       * of a Guice injector which in turn will upset Guice.
85       */
86      @Override
87      protected Injector getInjector()
88      {
89          if (guiceModule == null) {
90              throw new IllegalStateException("Never found the Guice Module to use!");
91          }
92  
93          log.info("Returning injector from " + guiceModule.getClass().getName());
94          return Guice.createInjector(Stage.PRODUCTION, guiceModule);
95      }
96  
97      /**
98       * This method can be called by classes extending SetupServer to retrieve
99       * the actual injector. This requires some inside knowledge on where it is
100      * stored, but the actual key is not visible outside the guice packages.
101      */
102     Injector injector(final ServletContextEvent event)
103     {
104         return (Injector) event.getServletContext().getAttribute(Injector.class.getName());
105     }
106 }
